```python
"""Typed data models for introspection results."""

from __future__ import annotations

from typing import Any

from pydantic import BaseModel, Field


class FeatureInfo(BaseModel):
    name: str
    dtype: str | None = None
    source: str | None = None
    description: str | None = None
    metadata: dict[str, Any] = Field(default_factory=dict)


class DataSourceInfo(BaseModel):
    name: str
    source_type: str
    fields: list[str] = Field(default_factory=list)
    filters: dict[str, Any] = Field(default_factory=dict)
    description: str | None = None
    metadata: dict[str, Any] = Field(default_factory=dict)


class ThresholdInfo(BaseModel):
    name: str
    value: float | str
    segment: str | None = None
    operator: str | None = None
    description: str | None = None


class ComponentInfo(BaseModel):
    """Maps to a ComponentNode in the version's tree."""

    path: str
    node_type: str
    metadata: dict[str, Any] = Field(default_factory=dict)


class IntrospectionResult(BaseModel):
    """Typed contract returned by all introspectors."""

    # Identity
    introspector: str
    framework: str | None = None

    # Model architecture
    algorithm: str | None = None
    hyperparameters: dict[str, Any] = Field(default_factory=dict)

    # Inputs
    features: list[FeatureInfo] = Field(default_factory=list)
    data_sources: list[DataSourceInfo] = Field(default_factory=list)

    # Outputs & decision logic
    thresholds: list[ThresholdInfo] = Field(default_factory=list)
    metrics: dict[str, float] = Field(default_factory=dict)
    output_description: str | None = None

    # Operational
    execution_schedule: str | None = None
    lookback_window: str | None = None

    # Structure mapping
    components: list[ComponentInfo] = Field(default_factory=list)

    # Escape hatch
    metadata: dict[str, Any] = Field(default_factory=dict)
```

```python
"""Introspector registry with lazy entry point discovery."""

from __future__ import annotations

from typing import Any

from model_ledger.core.exceptions import NoIntrospectorError
from model_ledger.introspect.protocol import Introspector

ENTRY_POINT_GROUP = "model_ledger.introspectors"

_registry: IntrospectorRegistry | None = None


class IntrospectorRegistry:
    def __init__(self) -> None:
        self._introspectors: list[Introspector] = []
        self._discovered = False

    def register(self, introspector: Introspector) -> None:
        """Register an introspector. Deduplicates by name — later registration wins.
        Prepends so manually registered introspectors take priority."""
        self._introspectors = [i for i in self._introspectors if i.name != introspector.name]
        self._introspectors.insert(0, introspector)

    def find(self, obj: Any) -> Introspector:
        """Find first introspector that can handle obj."""
        self._ensure_discovered()
        for intro in self._introspectors:
            if intro.can_handle(obj):
                return intro
        raise NoIntrospectorError(type(obj))

    def get_by_name(self, name: str) -> Introspector:
        """Get introspector by name."""
        self._ensure_discovered()
        for intro in self._introspectors:
            if intro.name == name:
                return intro
        raise NoIntrospectorError(name)

    def _ensure_discovered(self) -> None:
        """Lazy-load entry points on first use."""
        if self._discovered:
            return
        self._discovered = True
        try:
            from importlib.metadata import entry_points

            eps = entry_points(group=ENTRY_POINT_GROUP)
            for ep in sorted(eps, key=lambda e: e.name):
                try:
                    cls = ep.load()
                    instance = cls()
                    # Append (not prepend) so entry points have lower priority than manual registrations
                    if not any(i.name == instance.name for i in self._introspectors):
                        self._introspectors.append(instance)
                except Exception:
                    pass  # Skip broken entry points silently
        except Exception:
            pass


def get_registry() -> IntrospectorRegistry:
    """Get or create the global introspector registry."""
    global _registry
    if _registry is None:
        _registry = IntrospectorRegistry()
    return _registry


def reset_registry() -> None:
    """Reset the global registry. Used in tests."""
    global _registry
    _registry = None
```

```python
"""sklearn introspector — extracts metadata from scikit-learn estimators."""

from __future__ import annotations

from typing import Any

from model_ledger.introspect.models import (
    ComponentInfo,
    FeatureInfo,
    IntrospectionResult,
)


class SklearnIntrospector:
    name = "sklearn"

    def can_handle(self, obj: Any) -> bool:
        try:
            from sklearn.base import BaseEstimator

            return isinstance(obj, BaseEstimator)
        except ImportError:
            return False

    def introspect(self, obj: Any) -> IntrospectionResult:
        from sklearn.pipeline import Pipeline

        if isinstance(obj, Pipeline):
            return self._introspect_pipeline(obj)
        return self._introspect_estimator(obj)

    def _introspect_estimator(self, obj: Any) -> IntrospectionResult:
        features = []
        if hasattr(obj, "feature_names_in_"):
            features = [FeatureInfo(name=str(n)) for n in obj.feature_names_in_]

        return IntrospectionResult(
            introspector=self.name,
            framework="scikit-learn",
            algorithm=type(obj).__name__,
            hyperparameters=obj.get_params(deep=False),
            features=features,
        )

    def _introspect_pipeline(self, pipe: Any) -> IntrospectionResult:
        components = []
        final_estimator = None

        for step_name, step_obj in pipe.steps:
            cls_name = type(step_obj).__name__
            is_final = step_obj is pipe.steps[-1][1]
            node_type = "algorithm" if is_final else "preprocessor"
            components.append(
                ComponentInfo(
                    path=f"Processing/{cls_name}",
                    node_type=node_type,
                    metadata=step_obj.get_params(deep=False),
                )
            )
            if is_final:
                final_estimator = step_obj

        features = []
        if hasattr(pipe, "feature_names_in_"):
            features = [FeatureInfo(name=str(n)) for n in pipe.feature_names_in_]

        algorithm = type(final_estimator).__name__ if final_estimator else "Pipeline"

        return IntrospectionResult(
            introspector=self.name,
            framework="scikit-learn",
            algorithm=algorithm,
            hyperparameters=final_estimator.get_params(deep=False) if final_estimator else {},
            features=features,
            components=components,
        )
```

```python
"""model-ledger CLI — Typer app with Rich formatting."""

from __future__ import annotations

import json
import os
import sys
from typing import Optional

import typer
from rich.console import Console
from rich.table import Table

from model_ledger import Inventory
from model_ledger.validate.engine import validate

app = typer.Typer(name="model-ledger", help="Model inventory and governance CLI.")
console = Console()


def _get_db(db: str | None) -> str:
    if db:
        return db
    return os.environ.get("MODEL_LEDGER_DB", "inventory.db")


@app.command(name="list")
def list_models(
    db: Optional[str] = typer.Option(None, "--db", help="Path to inventory database"),
    format: str = typer.Option("table", "--format", help="Output format: table or json"),
):
    """List all models in the inventory."""
    inv = Inventory(db_path=_get_db(db))
    models = inv.list_models()

    if format == "json":
        typer.echo(json.dumps([m.model_dump(mode="json") for m in models], indent=2, default=str))
        return

    if not models:
        console.print("[dim]No models in inventory.[/dim]")
        return

    table = Table(title="Models")
    table.add_column("Name")
    table.add_column("Owner")
    table.add_column("Tier")
    table.add_column("Status")
    table.add_column("Type")

    for m in models:
        table.add_row(m.name, m.owner, m.tier.value, m.status.value, m.model_type.value)

    console.print(table)


@app.command()
def show(
    model_name: str = typer.Argument(..., help="Model name"),
    version: Optional[str] = typer.Option(None, "--version", "-v", help="Version to show"),
    db: Optional[str] = typer.Option(None, "--db", help="Path to inventory database"),
    format: str = typer.Option("table", "--format", help="Output format: table or json"),
):
    """Show model details."""
    inv = Inventory(db_path=_get_db(db))
    try:
        model = inv.get_model(model_name)
    except Exception as e:
        console.print(f"[red]Error:[/red] {e}")
        raise typer.Exit(1)

    if format == "json":
        typer.echo(model.model_dump_json(indent=2))
        return

    console.print(f"[bold]{model.name}[/bold]")
    console.print(f"  Owner: {model.owner}")
    console.print(f"  Tier: {model.tier.value}")
    console.print(f"  Status: {model.status.value}")
    console.print(f"  Type: {model.model_type.value}")
    console.print(f"  Purpose: {model.intended_purpose}")

@app.command(name="introspect")
def introspect_cmd(
    path: str = typer.Argument(..., help="Path to serialized model artifact"),
    introspector_name: Optional[str] = typer.Option(None, "--introspector", help="Introspector name"),
    allow_pickle: bool = typer.Option(False, "--allow-pickle", help="Allow loading pickle files (security risk)"),
    format: str = typer.Option("table", "--format"),
):
    """Introspect a serialized model artifact."""
    import pickle
    from pathlib import Path as P

    file_path = P(path)
    if not file_path.exists():
        console.print(f"[red]File not found: {path}[/red]")
        raise typer.Exit(1)

    suffix = file_path.suffix.lower()

    if suffix in (".pkl", ".pickle") and not allow_pickle:
        console.print("[red]Pickle files require --allow-pickle flag (arbitrary code execution risk)[/red]")
        raise typer.Exit(1)

    if suffix in (".pkl", ".pickle"):
        console.print("[yellow]Warning: Loading pickle file — this can execute arbitrary code.[/yellow]")
        with open(file_path, "rb") as f:
            obj = pickle.load(f)  # noqa: S301
    elif suffix == ".joblib":
        try:
            import joblib
            obj = joblib.load(file_path)
        except ImportError:
            console.print("[red]joblib not installed. Run: pip install joblib[/red]")
            raise typer.Exit(1)
    else:
        console.print(f"[red]Unsupported file format: {suffix}[/red]")
        raise typer.Exit(1)

    import model_ledger
    result = model_ledger.introspect(obj, introspector=introspector_name)

    if format == "json":
        typer.echo(result.model_dump_json(indent=2))
    else:
        console.print(f"[bold]Introspector:[/bold] {result.introspector}")
        console.print(f"[bold]Framework:[/bold] {result.framework or '-'}")
        console.print(f"[bold]Algorithm:[/bold] {result.algorithm or '-'}")
        if result.features:
            console.print(f"[bold]Features:[/bold] {len(result.features)}")
        if result.hyperparameters:
            console.print(f"[bold]Hyperparameters:[/bold] {result.hyperparameters}")
        if result.metrics:
            console.print(f"[bold]Metrics:[/bold] {result.metrics}")
        if result.thresholds:
            console.print(f"[bold]Thresholds:[/bold] {len(result.thresholds)}")
```

```python
"""Audit pack export — self-contained compliance artifacts."""

from __future__ import annotations

import json
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

import model_ledger
from model_ledger.sdk.inventory import Inventory
from model_ledger.validate.engine import validate


def export_audit_pack(
    *,
    inventory: Inventory,
    model_name: str,
    version: str | None = None,
    format: str = "html",
    output_path: str,
) -> None:
    """Export an audit pack for a model."""
    data = _gather_data(inventory, model_name, version)

    if format == "json":
        _write_json(data, output_path)
    elif format == "html":
        _write_html(data, output_path)
    elif format == "markdown":
        _write_markdown(data, output_path)
    else:
        raise ValueError(f"Unknown format: {format}. Use 'json', 'html', or 'markdown'.")


def _gather_data(inventory: Inventory, model_name: str, version_str: str | None) -> dict[str, Any]:
    model = inventory.get_model(model_name)
    model_data = model.model_dump(mode="json")

    # Find version
    version_data = None
    version_obj = None
    if version_str:
        version_obj = inventory.get_version(model_name, version_str)
    else:
        # Find latest version via backend
        all_versions = inventory._backend.list_versions(model_name)
        version_obj = all_versions[-1] if all_versions else None

    if version_obj:
        version_data = version_obj.model_dump(mode="json")

    # Validation
    validation_data = None
    if version_obj:
        try:
            result = validate(model, version_obj, profile="sr_11_7")
            validation_data = {
                "profile": result.profile,
                "passed": result.passed,
                "violations": [
                    {"rule_id": v.rule_id, "severity": v.severity, "message": v.message, "suggestion": v.suggestion}
                    for v in result.violations
                ],
            }
        except Exception:
            pass

    # Audit trail
    audit_events = inventory.get_audit_log(model_name)
    audit_data = [e.model_dump(mode="json") for e in audit_events]

    return {
        "model": model_data,
        "version": version_data,
        "validation": validation_data,
        "audit_trail": audit_data,
        "exported_at": datetime.now(timezone.utc).isoformat(),
        "model_ledger_version": model_ledger.__version__,
    }


def _write_json(data: dict, output_path: str) -> None:
    Path(output_path).write_text(json.dumps(data, indent=2, default=str))


def _write_html(data: dict, output_path: str) -> None:
    from model_ledger.export.templates import HTML_TEMPLATE

    sections = []
    m = data["model"]
    sections.append(f"<h1>{m['name']}</h1>")
    sections.append("<h2>Model Card</h2>")
    sections.append("<table>")
    for key in ["name", "owner", "tier", "status", "model_type", "intended_purpose", "business_unit"]:
        val = m.get(key, "-")
        if val:
            sections.append(f"<tr><th>{key.replace('_', ' ').title()}</th><td>{val}</td></tr>")
    sections.append("</table>")

    if data["version"]:
        v = data["version"]
        sections.append(f"<h2>Version: {v['version']}</h2>")
        sections.append("<table>")
        for key in ["status", "methodology_approach", "run_frequency", "training_target"]:
            val = v.get(key)
            if val:
                sections.append(f"<tr><th>{key.replace('_', ' ').title()}</th><td>{val}</td></tr>")
        sections.append("</table>")

    if data["validation"]:
        vr = data["validation"]
        status_cls = "pass" if vr["passed"] else "fail"
        status_txt = "PASS" if vr["passed"] else "FAIL"
        sections.append(f'<h2>Validation: <span class="{status_cls}">{status_txt}</span></h2>')
        if vr["violations"]:
            sections.append("<table><tr><th>Rule</th><th>Severity</th><th>Message</th></tr>")
            for v in vr["violations"]:
                sections.append(f"<tr><td>{v['rule_id']}</td><td>{v['severity']}</td><td>{v['message']}</td></tr>")
            sections.append("</table>")

    if data["audit_trail"]:
        sections.append("<h2>Audit Trail</h2>")
        for e in data["audit_trail"]:
            ts = str(e.get("timestamp", ""))[:19]
            sections.append(
                f'<div class="audit-event"><strong>{e["action"]}</strong> by {e["actor"]} at {ts}</div>'
            )

    content = "\n".join(sections)
    html = HTML_TEMPLATE.format(
        model_name=m["name"],
        content=content,
        version=data["model_ledger_version"],
    )
    Path(output_path).write_text(html)


def _write_markdown(data: dict, output_path: str) -> None:
    lines = []
    m = data["model"]
    lines.append(f"# {m['name']}")
    lines.append("")
    lines.append("## Model Card")
    lines.append("")
    lines.append(f"| Field | Value |")
    lines.append(f"|---|---|")
    for key in ["owner", "tier", "status", "model_type", "intended_purpose"]:
        val = m.get(key, "-")
        if val:
            lines.append(f"| {key.replace('_', ' ').title()} | {val} |")

    if data["version"]:
        v = data["version"]
        lines.append("")
        lines.append(f"## Version: {v['version']}")
        lines.append("")
        for key in ["status", "methodology_approach", "run_frequency", "training_target"]:
            val = v.get(key)
            if val:
                lines.append(f"- **{key.replace('_', ' ').title()}**: {val}")

    if data["validation"]:
        vr = data["validation"]
        status = "PASS" if vr["passed"] else "FAIL"
        lines.append("")
        lines.append(f"## Validation: {status}")
        if vr["violations"]:
            lines.append("")
            lines.append("| Rule | Severity | Message |")
            lines.append("|---|---|---|")
            for v in vr["violations"]:
                lines.append(f"| {v['rule_id']} | {v['severity']} | {v['message']} |")

    if data["audit_trail"]:
        lines.append("")
        lines.append("## Audit Trail")
        lines.append("")
        for e in data["audit_trail"]:
            ts = str(e.get("timestamp", ""))[:19]
            lines.append(f"- **{e['action']}** by {e['actor']} at {ts}")

    lines.append("")
    lines.append(f"---\n*Generated by model-ledger v{data['model_ledger_version']}*")
    Path(output_path).write_text("\n".join(lines))
```
